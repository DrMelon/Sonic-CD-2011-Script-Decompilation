
//-------Aliases-------//
#alias 83: TYPE_GOLFMETERPIP

sub ObjectMain
    Object.XPos=Snolf.BarPosX
    TempValue0=Snolf.MeterX
    TempValue0*=10000
    Object.XPos+=TempValue0
    Object.YPos=Snolf.BarPosY
    TempValue0=Snolf.MeterY
    TempValue0*=10000
    Object.YPos+=TempValue0
    
    if Snolf.ModeState == 0 
        Object.Type=TypeName[BlankObject]
    endif
    if Snolf.ModeState == 2
        Object.XPos=Snolf.BarPosX
    endif
endsub

sub ObjectDraw
	DrawSpriteXY(0,Object.XPos,Object.YPos)
   
endsub


sub ObjectStartup
	LoadSpriteSheet("Global/Items.gif")
	SpriteFrame(-8,-8,16,16,1,1)
endsub

sub RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
endsub
